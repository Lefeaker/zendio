🔍 DOM 结构调试工具

使用说明：
1. 在通义千问或 DeepSeek 的对话页面打开此工具（在新标签页打开）
2. 切换回对话页面
3. 打开浏览器控制台（F12）
4. 复制下面的代码并在控制台运行
5. 查看输出结果
📋 步骤 1: 检查页面信息
// 检查当前页面 console.log('=== 页面信息 ==='); console.log('URL:', location.href); console.log('域名:', location.hostname); console.log('标题:', document.title); // 检查是否是通义千问 if (/tongyi/.test(location.href)) { console.log('✅ 这是通义千问页面'); } // 检查是否是 DeepSeek if (/deepseek/.test(location.href)) { console.log('✅ 这是 DeepSeek 页面'); }
📋 步骤 2: 查找消息容器
// 查找所有可能的消息容器 console.log('\n=== 查找消息容器 ==='); const selectors = [ 'article', '[class*="message"]', '[class*="Message"]', '[class*="chat"]', '[class*="Chat"]', '[data-role]', '[data-type]', '[class*="conversation"]', '[class*="dialog"]', ]; selectors.forEach(selector => { const elements = document.querySelectorAll(selector); if (elements.length > 0) { console.log(`✅ 找到 ${elements.length} 个元素: ${selector}`); console.log(' 第一个元素:', elements[0]); console.log(' 类名:', elements[0].className); console.log(' 属性:', Array.from(elements[0].attributes).map(a => `${a.name}="${a.value}"`).join(', ')); } });
📋 步骤 3: 分析消息结构（通义千问）
// 通义千问专用分析 console.log('\n=== 通义千问消息结构 ==='); // 查找所有 article 元素 const articles = document.querySelectorAll('article'); console.log(`找到 ${articles.length} 个 article 元素`); articles.forEach((article, index) => { console.log(`\n--- Article ${index + 1} ---`); console.log('类名:', article.className); console.log('data-role:', article.getAttribute('data-role')); console.log('data-type:', article.getAttribute('data-type')); // 查找角色指示器 const userIndicators = article.querySelectorAll('[class*="user"], [class*="User"]'); const assistantIndicators = article.querySelectorAll('[class*="assistant"], [class*="Assistant"], [class*="bot"]'); console.log('用户指示器:', userIndicators.length); console.log('助手指示器:', assistantIndicators.length); // 获取文本内容 const text = article.textContent?.trim().substring(0, 100); console.log('内容预览:', text); });
📋 步骤 4: 分析消息结构（DeepSeek）
// DeepSeek 专用分析 console.log('\n=== DeepSeek 消息结构 ==='); // 查找所有可能的消息容器 const possibleContainers = [ '[class*="message"]', '[class*="Message"]', '[class*="chat-message"]', 'article', '[data-role]', ]; possibleContainers.forEach(selector => { const elements = document.querySelectorAll(selector); if (elements.length > 0) { console.log(`\n找到 ${elements.length} 个元素: ${selector}`); // 分析前3个元素 elements.forEach((el, index) => { if (index < 3) { console.log(`\n 元素 ${index + 1}:`); console.log(' 类名:', el.className); console.log(' data-role:', el.getAttribute('data-role')); console.log(' data-type:', el.getAttribute('data-type')); console.log(' 内容预览:', el.textContent?.trim().substring(0, 50)); } }); } });
📋 步骤 5: 完整 DOM 树分析
// 完整 DOM 树分析 console.log('\n=== 完整 DOM 树分析 ==='); function analyzeElement(element, depth = 0) { const indent = ' '.repeat(depth); const tag = element.tagName.toLowerCase(); const className = element.className ? ` class="${element.className}"` : ''; const dataRole = element.getAttribute('data-role') ? ` data-role="${element.getAttribute('data-role')}"` : ''; const dataType = element.getAttribute('data-type') ? ` data-type="${element.getAttribute('data-type')}"` : ''; console.log(`${indent}<${tag}${className}${dataRole}${dataType}>`); // 只分析前3层 if (depth < 3) { Array.from(element.children).forEach(child => { analyzeElement(child, depth + 1); }); } } // 查找主要内容区域 const mainContent = document.querySelector('main') || document.querySelector('[role="main"]') || document.body; console.log('分析主要内容区域:'); analyzeElement(mainContent);
📋 步骤 6: 导出完整信息
// 导出完整信息到剪贴板 console.log('\n=== 导出完整信息 ==='); const info = { url: location.href, hostname: location.hostname, title: document.title, articles: [], messages: [], }; // 收集所有 article document.querySelectorAll('article').forEach((article, index) => { info.articles.push({ index: index, className: article.className, dataRole: article.getAttribute('data-role'), dataType: article.getAttribute('data-type'), textPreview: article.textContent?.trim().substring(0, 100), childrenCount: article.children.length, innerHTML: article.innerHTML.substring(0, 200), }); }); // 收集所有可能的消息 document.querySelectorAll('[class*="message"], [class*="Message"]').forEach((msg, index) => { info.messages.push({ index: index, className: msg.className, dataRole: msg.getAttribute('data-role'), dataType: msg.getAttribute('data-type'), textPreview: msg.textContent?.trim().substring(0, 100), }); }); console.log('收集到的信息:', info); console.log('\n请复制上面的信息并发送给开发者'); // 尝试复制到剪贴板 const jsonStr = JSON.stringify(info, null, 2); navigator.clipboard.writeText(jsonStr).then(() => { console.log('✅ 信息已复制到剪贴板！'); }).catch(err => { console.log('❌ 复制失败，请手动复制上面的信息'); });
